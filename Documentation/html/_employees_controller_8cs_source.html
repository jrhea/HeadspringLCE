67  public ActionResult Index(string category, string currentFilter, string searchString, int? page)
68  {
69  ViewResult result;
70  Type type = typeof(EmployeeData.Employee);
71  int pageSize = 3;
72  int pageNumber = (page ?? 1);
73  if (searchString != null)
74  {
75  page = 1;
76  }
77  else
78  {
79  searchString = currentFilter;
80  }
81 
82  ViewBag.CurrentFilter = searchString;
83  ViewBag.Category = category;
84 
85  //This is just a clever way of using Reflection and Extension methods
86  //in order to not have to maintain a list of "Searchable" properties
87  if (_categoryDisplayNames == null)
88  {
89  //Find properties of Employee that have SearchableAttribute with Value = true
90  List<PropertyInfo> propList1 = type.getMatchingProperties<Searchable,Boolean>(true);
91  //Find properties of Employee that have DisplayAttribute
92  List<PropertyInfo> propList2 = type.getMatchingProperties<DisplayNameAttribute>();
93  //Find the intersection of the two lists
94  List<PropertyInfo> categoryPropInfos = propList1.Intersect(propList2).ToList<PropertyInfo>();
95  //Get list of user friendly names for properties
96  _categoryDisplayNames = categoryPropInfos.Select(x => x.GetCustomAttribute<DisplayNameAttribute>().DisplayName).ToList<String>();
97  }
98 
99 
100  ViewBag.Category = new SelectList(_categoryDisplayNames);
101 
102  var employees = _esc.GetAllEmployees().AsQueryable();
103 
104  //Find the Employee objects with property that matches criteria
105  if (!String.IsNullOrEmpty(category) && !String.IsNullOrEmpty(searchString))
106  {
107  List<Employee> matches = new List<Employee>();
108  PropertyInfo info = type.getMatchingProperties<DisplayNameAttribute, String>(category, "DisplayName").First();
109  foreach (Employee employee in employees)
110  {
111  if(info.GetValue(employee).ToString().Contains(searchString))
112  {
113  matches.Add(employee);
114  }
115  }
116  result = View(matches.ToPagedList(pageNumber,pageSize));
117  }
118  else
119  {
120  Session["ModelHelper"] = _modelHelper;
121  result = View(employees.ToPagedList(pageNumber,pageSize));
122  }
123 
124  return result;
125  }
